    • Jetpack version
      sudo apt-cache show nvidia-jetpack

      head -n1 /etc/nv_tegra_release

      dpkg -l|grep nvinfer
    • CUDA
      nvcc --version
    • cuDNN
      dpkg -l|grep cudnn
    • Python
      python --version
